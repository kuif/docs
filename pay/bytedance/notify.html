# 例子

$pay = new \fengkui\Pay\Bytedance($bytedanceConfig);

// 支付回调
try {
    $data = $pay->notify(); // 验签并获取支付回调相关参数
    if ($data) { // 支付成功
        return $pay->success(); // 异步通知
    }
} catch (\Exception $e) {
    $e->getMessage(); // 错误信息
}
