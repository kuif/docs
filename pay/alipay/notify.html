支付回调

$pay = new \fengkui\Pay\Alipay($alipayConfig);

// 支付回调
try {
    $data = $pay->notify(); // 验签并获取支付回调相关参数
    if ($data) { // 支付成功
        echo "success";die; // 异步通知
    }
} catch (\Exception $e) {
    $e->getMessage(); // 错误信息
}
// 当回调数据为空时，请打印错误信息
// $pay->notify(false); 不验证签名，直接返回支付信息
