换取授权访问令牌

$pay = new \fengkui\Pay\Alipay($alipayConfig);
$result = $pay->getToken(); // 直接跳转(静默授权：auth_base)
$result = $pay->getToken(false); // 直接跳转(主动授权：auth_user)
echo $result;
